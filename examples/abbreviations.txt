[ i += 1 ] => [ i = i + 1 ]
[ i += 5 ] => [ i = i + 5 ]

[ i -= 1 ] => [ i = i - 1 ]
[ i -= 5 ] => [ i = i - 5 ]

[ i *= 1 ] => [ i = i * 1 ]
[ i *= 5 ] => [ i = i * 5 ]

[ i /= 1 ] => [ i = i / 1 ]
[ i /= 5 ] => [ i = i / 5 ]

[ i |= 1 ] => [ i = i | 1 ]
[ i |= 5 ] => [ i = i | 5 ]

[ i &= 1 ] => [ i = i & 1 ]
[ i &= 5 ] => [ i = i & 5 ]

[ i++ ] => [ i += 1 ] => [ i = i + 1 ]
[ i-- ] => [ i -= 1 ] => [ i = i - 1 ]
